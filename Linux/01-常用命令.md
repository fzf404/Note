## 基础命令

```bash
pwd # 当前目录

ls <path> # 列出全部目录
  -lah # 详细信息|全部|文件大小

cp <source> <target> # 复制目录
mv <source> <target> # 移动目录 
  -riuv # 递归|覆盖提示|不覆盖|实时信息

mkdir -R <path> # 新建目录
rm <path> # 删除文件/目录
  -if # 删除提示|强制删除 
    
ln <source> <target> # 创建链接
  -s # 创建软链接

man <command> # 命令简介
```

## 工具

```bash
# 时间
date # 当前日期
# 格式化输出
echo $(date "+%Y_%m_%d_%H_%M_%S")

cal # 日历

# 文件
file <file> # 文件信息
cat <file> # 文件内容
less <file> # 文件内容

# 搜索
grep <content> # 搜索内容
  -niv # 行数|忽略大小写|取反
grep err.log -e painc
    
find <path> -name <file> # 计算机内搜索内容
  -name # 文件名
  -size # 大小
    +15K # 大于15KB
    -1M # 小于1MB
  -type # 文件类型
  -exec # 对搜索出的文件运行命令
    rm {} \; # 删除

# 归档及压缩
gzip -d # 解压缩
tar # 归档|解档|压缩|解压
  -cvf # 归档
  -xvf # 解档
  -zcvf # 压缩
  -zxvf # 解压

# 信息
type <command> # 命令详情
which <command> # 命令程序位置
whatis <command> # 命令简洁说明
alias <alias>=<command> # 未命令创建别名
```

## 软件安装

```bash
# 搜索
apt search <package>
# 安装
apt install <package>
# 卸载
apt remove <package>
# 彻底卸载
apt-get remove --purge

# 安装软件包
dpkg -i *.deb
```
