```python
import pymysql

# 连接数据库
conn = pymysql.connect(host='127.0.0.1',
                       user='root',
                       password='password',
                       database='temp',
                       cursorclass=pymysql.cursors.DictCursor)

with conn.cursor() as cursor:
    # Create a new record
    sql = "INSERT INTO `User` (`name`, `age`, `sex`) VALUES (%s, %s, %s)"
    cursor.execute(sql, (name, age, sex))
    
conn.commit()
conn.close()
```
