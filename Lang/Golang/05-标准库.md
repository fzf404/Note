## Log

```go
// 种类
log.Print("基础的日志")
log.Println("换行的日志")
log.Printf("%s的日志", "f")

// 日志文件
fileName := "debug.log"
// 创建文件
logFile,err  := os.Create(fileName)  
defer logFile.Close()      
if err != nil {  
  log.Fatalln("Error: open log file!")  
}
// 文件句柄，日志前缀，配置日志模式
debugLog := log.New(logFile,"",log.Lshortfile)

// Log分类
debugLog.Print("打印日志并继续执行")
debugLog.Fatal("打印日志并终止执行")
debugLog.Panic("打印日志并停止执行")
```
