## 联合体

> 每个实例只能存储一个声明的变量
>
> 最多只有一个成员有效

```c
#include <stdio.h>

union TestUnion
{
    char x;
    int y;
};

int main(int argc, char const *argv[])
{

    union TestUnion test;
    test.x = 'a';
    test.y = 404;
    printf("size: %d\nValue: %d", sizeof(test),test.y);
    // size: 4
	// Value: 404

    return 0;
}
```

### 结构体指针

```c
#include <stdio.h>

struct point* StructInput(struct point *p);

struct point {
	int x;
	int y;
};

int main()
{
	struct point place;
	// StructInput(&place);
    printf("x is: %d\n",(*StructInput(&place)).x);
	printf("place is (%d,%d).",place.x,place.y);
}

struct point* StructInput(struct point *p)
{
	printf("Input x,y: ");
	scanf("%d,%d",&p->x,&(*p).y);
	return p;
}

/*

x is: 3
place is (3,4).
```

## 可变数组

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct
{
	int *array;
	int size;
} Array;

Array array_creat(int init_size)
{
	Array a;
	a.size = init_size;
	a.array = (int *)malloc(sizeof(int) * a.size);
	return a;
}

void array_free(Array *a)
{
	free(a->array);
	a->array = NULL;
	a->size = 0;
}


Array array_icrease(Array *a, int increase)
{
	Array new = array_creat(a->size+increase);
	for(int i; i<a->size; i++)
	{
		new.array[i] = a->array[i];
	}
	array_free(a);
	return new;
}


int array_size(Array *a)
{
	return a->size;
}

int array_get(Array *a, int index)
{
	return a->array[index];
}

Array array_input(Array *a, int index, int num)
{
	a->array[index] = num;
	return *a;
}

int main()
{
	Array a = array_creat(100);
	// 批量赋值
	for (int i; i < 100; i++)
		array_input(&a, i, i);
	// 增加大小
	a = array_icrease(&a,20);
	// 打印信息
	printf("ArraySize is %d.\n", array_size(&a));
	printf("Array[10] is %d.\nArray[110] is %d.", array_get(&a, 10), array_get(&a, 110));


	return 0;
}

/*
ArraySize is 120.
Array[10] is 10.
Array [110] is 0.
*/
```

## 链表

```c
#include <stdio.h>
#include <stdlib.h>

typedef struct LNode
{
	int data;
	struct LNode *next;
} LNode, *LinkList;

LinkList CreateList(int n);
void print(LinkList h);
int main()
{
	LinkList Head = NULL;
	int n;

	scanf("%d", &n);
	Head = CreateList(n);

	printf("Num is: \n");
	print(Head);

	printf("\n\n");
	system("pause");
	return 0;
}
LinkList CreateList(int n)
{
	LinkList L, p, q;
	int i;
	L = (LNode *)malloc(sizeof(LNode));
	if (!L)
		return 0;
	L->next = NULL;
	q = L;
	for (i = 1; i <= n; i++)
	{
		p = (LinkList)malloc(sizeof(LNode));
		printf("Input No_%d Value: ", i);
		scanf("%d", &(p->data));
		p->next = NULL;
		q->next = p;
		q = p;
	}
	return L;
}
void print(LinkList h)
{
	LinkList p = h->next;
	while (p != NULL)
	{
		printf("%d ", p->data);
		p = p->next;
	}
}
```
