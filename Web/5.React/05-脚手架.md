> `App.js`

```react
import React from 'react';
import './App.scss'

class App extends React.Component {
  state = {
    val: '',
    list: []
  }

  handleChange = (event) => {
    let val = event.target.value;
    this.setState({ val })
  }

  handleAdd = (event) => {
    let {val,list} = this.state;
    list.push(val);
    this.setState({list})
  }

  render() {
    const { val,list } = this.state
    return <div>
      <h1>React Test</h1>
      <input type="text" val={val} onChange={this.handleChange} />
      <button onClick={this.handleAdd}>Add</button>
      <br></br>
      <ul>
        {
          list.map(item =><li>{item}</li>)
        }
      </ul>
    </div>
  }
}

export default App;
```
