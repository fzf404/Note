## 函数

```typescript
// (参数名:参数类型):返回值类型
function WYN(name:string):void {
  console.log(name);
}

// type 声明函数
type AddFun = (a:number, b:number)=>number;
// interface 声明函数
interface AddFun {
 	 (a:number, b:number):number   
}
// 函数实现
let add:AddFun = function (x, y) {
    return x + y;
};
let res = add(30, 20);

// 正常参数 默认参数 可选参数 
function add(x:number, y:number=10, z?:number):number {
  return x + y + (z ? z : 0);
}
// 更多参数
...args:number[]

// 箭头函数	传参	f
let getArray = (value:any, items:number = 5):any[]=>{
  return new Array(items).fill(value);
};
```

## 接口

```typescript
interface Todo {
  content: string
  // 只读
  readonly user: string
  // ?可选
  time?: string
  isCompleted: boolean
}

// 创建实例
let todo01: Todo = {
    content: 'Learn TypeScript',
    user: 'fzf404',
    time: '2020-5-20 19:47',
    isCompleted: false
}

// 解构赋值
function getTodo({content,time,isCompleted}:Todo):void {
    console.log(`${isCompleted?'已完成':'未完成'}: ${content}\n${time}`)
}

getTodo(todo01)

// 索引签名
interface Setp {
    [propName:number]:string
}

let arr:Setp = {
    0: '1.查找官方文档',
    1: '2.在线TS编辑器实验',
    2: '3.使用TS开发React应用'
};

console.log(arr);

// 继承
interface TodoInfo extends Todo,Setp {
    doing: boolean
}

let todoinfo01:TodoInfo = {
    ...Todo,
    ...Setp,
    doing: false
}
```

## 类

```typescript
class Person {
    name: string;
    constructor(name: string) {
        this.name = name;
    }
    greet() {
        return `${this.name} say: hi~`
    }
}

let fzf = new Person('fzf');
console.log(fzf.greet())
// fzf say: hi~

// 继承
class Student extends Person {
    private major: string;
    constructor(name: string, major: string) {
        super(name)
        this.major = major;
    }
    studentGreet() {
        return `${this.major} manjor's ${this.name} say hi~.`
    }
}

let fzf = new Student('fzf', 'anime')
console.log(fzf.studentGreet()) 
// anime manjor's fzf say hi~.
```
