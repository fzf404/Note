## 新东西

```vue
<template>
  <div id="app">
    <h2>{{ name }}</h2>
    <p>{{ age }}</p>
    <p>{{ increase }}</p>
    <button @click="addAge()">Age++</button>
    <button @click="stopWatch()">StopWatch</button>
  </div>
</template>
  
<script>
import { ref, toRefs, watch, onMounted, reactive } from "vue";
export default {
  setup() {
    // ref创建一个响应式的数据对象
    const name = ref("fzf");
    const age = ref(18);
    const increase = ref(0);

    // 使用reactive创建复杂数据类型
    // 读取时无需.value
    const user = reactive({ name: "fzf", age: age });

    // 增加年龄
    function addAge() {
      age.value++; // 改变或获取值使用.value
    }

    // 使用watch监听age修改
    function addIncrease() {
      increase.value++;
    }
    // 调用返回值可停止watch
    const stopWatch = watch(age, addIncrease);

    // 在 `mounted` 时调用 ` addAge`
    onMounted(addAge);

    // 返回后模板中可使用
    return {
      // 解构user
      ...toRefs(user),
      increase,
      addAge,
      stopWatch,
    };
  },
};
</script>
```
