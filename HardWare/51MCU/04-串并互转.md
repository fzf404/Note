# 串口与并口相互转换

## 74HC165(并转串)

```c
#include "reg51.h"
#include "intrins.h"

typedef unsigned char u8;
typedef unsigned int u16;

#define GPIO_LED P0

sbit IN_PL=P1^6;
sbit IN_Data=P1^7;
sbit SCK=P3^6;

u8 Read165()
{
	u8 indata = 0;
	u8 i;
	
	IN_PL = 0;
	_nop_();
	IN_PL = 1;
	_nop_();
	for(i=0;i<8;i++)
	{
		indata = indata<<1;
		SCK = 0;
		_nop_();
		
		indata |= IN_Data ;
		SCK = 1;
	}
	return indata;
}


void main()
{
	u8 value165;
	GPIO_LED = 0;
	while(1)
	{
		value165=Read165();
		if(value165!=0xff)
		{
			GPIO_LED = ~value165;
		}
	}
}
```

## 74HC595(串转并)

```c
#include "reg51.h"
#include "intrins.h"

typedef unsigned char u8;
typedef unsigned int u16;

#define GPIO_LED P0

sbit SRCLK=P3^6;
sbit RCLK=P3^5;
sbit SCK=P3^4;

void delay(u16 i)
{
	while(i--);
}


void hc595(u8 dat)
{
	u8 a;
	for(a=0;a<8;a++)
	{
		SCK = dat>>7;	//右移获取最高位
		dat<<=1;		//左移删掉以传输的数据
		
		SRCLK = 0;
		_nop_();
		_nop_();
		SRCLK = 1;
	}
	RCLK = 0;
	_nop_();
	_nop_();
	RCLK = 1;
}


void main()
{
	u8 ledNum;
	ledNum = 0x01;
	while(1)
	{
		hc595(ledNum);
		ledNum = _crol_(ledNum,1);
		delay(50000);
	}
}
```
