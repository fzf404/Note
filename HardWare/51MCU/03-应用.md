# 单片机应用

## 电子琴

```C
#include "reg51.h"

typedef unsigned char u8;
typedef unsigned int u16;

sbit beep=P0^5;

void delay(u16 i)
{
	while(i--);
}

u8 keypros()
{
	while(1)
	{
		if(P1<0xff)
		{
			delay(1000);
			switch(P1)
			{
				case(0x7f):return 110;break;
				case(0xbf):return 120;break;
				case(0xdf):return 140;break;
				case(0xef):return 160;break;
				case(0xf7):return 50;break;
				case(0xfb):return 60;break;
				case(0xfd):return 70;break;
				case(0xfe):return 90;break;
			}				
		}
	}	

}	

void be(u8 k)
{
	int i;
	for(i=0;i<=(180-k)*5;i++)
	{
		beep = ~beep;
		delay(k);
	}
}

void main()
{
	while(1)
	{
		be(keypros());
;	}
}
```

## 交通灯

```c
#include "reg51.h"

typedef unsigned char u8;
typedef unsigned int u16;

sbit LSA = P2 ^ 2;
sbit LSB = P2 ^ 3;
sbit LSC = P2 ^ 4;

#define GPIO_DIG P0
#define GPIO_TRAFFIC P1

sbit RED10 = P1 ^ 0;
sbit GREEN10 = P1 ^ 1;
sbit RED11 = P1 ^ 2;
sbit YELLOW11 = P1 ^ 3;
sbit GREEN11 = P1 ^ 4;

sbit RED00 = P3 ^ 0;
sbit GREEN00 = P3 ^ 1;
sbit RED01 = P1 ^ 5;
sbit YELLOW01 = P1 ^ 6;
sbit GREEN01 = P1 ^ 7;

u8 code smgduan[] = {0x3f, 0x06, 0x5b, 0x4f, 0x66, 0x6d, 0x7d, 0x07, 0x7f,
					 0x6f, 0x77, 0x7c, 0x39, 0x5e, 0x79, 0x71};

u8 seconds = 0;
u8 DisplayData[8];

void delay(u16 i)
{
	while (i--)
		;
}

void DigDisplay()
{
	u8 i;
	for (i = 0; i < 8; i++)
	{
		switch (i)
		{
		case 0:
			LSA = 0;
			LSB = 0;
			LSC = 0;
			break;
		case 1:
			LSA = 1;
			LSB = 0;
			LSC = 0;
			break;
		case 2:
			LSA = 0;
			LSB = 1;
			LSC = 0;
			break;
		case 3:
			LSA = 1;
			LSB = 1;
			LSC = 0;
			break;
		case 4:
			LSA = 0;
			LSB = 0;
			LSC = 1;
			break;
		case 5:
			LSA = 1;
			LSB = 0;
			LSC = 1;
			break;
		case 6:
			LSA = 0;
			LSB = 1;
			LSC = 1;
			break;
		case 7:
			LSA = 1;
			LSB = 1;
			LSC = 1;
			break;
		}
		GPIO_DIG = DisplayData[i];
		delay(100);
		GPIO_DIG = 0x00; // 清零
	}
}

void Timer0Init()
{
	TMOD |= 0x01;
	TH0 = 0xFC;
	TL0 = 0X18;
	ET0 = 1; // 打开中断
	EA = 1;	 // 打开总中断
	TR0 = 1;
}

void main()
{
	seconds = 1;
	Timer0Init();
	while (1)
	{
		if (seconds == 70)
		{
			seconds = 1;
		}
		if (seconds < 30)
		{
			DisplayData[0] = 0x00;
			DisplayData[1] = 0x00;
			DisplayData[2] = smgduan[(30 - seconds) / 10];
			DisplayData[3] = smgduan[(30 - seconds) % 10];

			DisplayData[4] = 0x00;
			DisplayData[5] = 0x00;
			DisplayData[6] = DisplayData[2];
			DisplayData[7] = DisplayData[3];
			DigDisplay();

			GPIO_TRAFFIC = 0xff;
			RED00 = 1;
			GREEN00 = 1;

			GREEN10 = 0;
			GREEN11 = 0;

			RED00 = 0;
			RED01 = 0;
		}
		else if (seconds < 35)
		{
			DisplayData[0] = 0x00;
			DisplayData[1] = 0x00;
			DisplayData[2] = smgduan[(35 - seconds) / 10];
			DisplayData[3] = smgduan[(35 - seconds) % 10];

			DisplayData[4] = 0x00;
			DisplayData[5] = 0x00;
			DisplayData[6] = DisplayData[2];
			DisplayData[7] = DisplayData[3];
			DigDisplay();

			GPIO_TRAFFIC = 0xff;
			RED00 = 1;
			GREEN00 = 1;

			RED10 = 0;
			YELLOW11 = 0;

			RED00 = 0;
			RED01 = 0;
		}
		else if (seconds < 65)
		{
			DisplayData[0] = 0x00;
			DisplayData[1] = 0x00;
			DisplayData[2] = smgduan[(65 - seconds) / 10];
			DisplayData[3] = smgduan[(65 - seconds) % 10];

			DisplayData[4] = 0x00;
			DisplayData[5] = 0x00;
			DisplayData[6] = DisplayData[2];
			DisplayData[7] = DisplayData[3];
			DigDisplay();

			GPIO_TRAFFIC = 0xff;
			RED00 = 1;
			GREEN00 = 1;

			RED10 = 0;
			RED11 = 0;

			GREEN00 = 0;
			GREEN01 = 0;
		}
		else
		{
			DisplayData[0] = 0x00;
			DisplayData[1] = 0x00;
			DisplayData[2] = smgduan[(70 - seconds) / 10];
			DisplayData[3] = smgduan[(70 - seconds) % 10];

			DisplayData[4] = 0x00;
			DisplayData[5] = 0x00;
			DisplayData[6] = DisplayData[2];
			DisplayData[7] = DisplayData[3];
			DigDisplay();

			GPIO_TRAFFIC = 0xff;
			RED00 = 1;
			GREEN00 = 1;

			RED10 = 0;
			RED11 = 0;

			YELLOW01 = 0;
			RED01 = 0;
		}
	}
}

void Time0() interrupt 1
{
	static u16 i;
	TH0 = 0xFC;
	TL0 = 0X18;
	i++;
	if (i == 1000)
	{
		i = 0;
		seconds++;
	}
}
```
